async function startAiScan() { if (!uploadedFilePath) { showToast('Tunggu gambar selesai terupload...', 'info'); return; } document.getElementById('scanningImgPreview').src = imageUrl(uploadedFilePath); goToStep('stepAiScanning'); try { const imgUrl = imageUrl(uploadedFilePath); const absoluteUrl = imgUrl.startsWith('http') ? imgUrl : window.location.origin + imgUrl; console.log('[OCR] Calling API with URL:', absoluteUrl); // Fetch dengan error handling lebih ketat const response = await fetch('/api/expenses?action=ocr', { method: 'POST', headers: { 'Content-Type': 'application/json', }, body: JSON.stringify({ url: absoluteUrl }) }); console.log('[OCR] Response status:', response.status); const contentType = response.headers.get('content-type'); console.log('[OCR] Content-Type:', contentType); // Cek response status if (!response.ok) { const errorText = await response.text(); console.error('[OCR] Error response:', errorText); throw new Error(`API Error ${response.status}: ${errorText.substring(0, 100)}`); } // Cek content type if (!contentType || !contentType.includes('application/json')) { const text = await response.text(); console.error('[OCR] Invalid content type. Response:', text); throw new Error(`Invalid response type: ${contentType}. Expected JSON.`); } const res = await response.json(); console.log('[OCR] Parsed response:', res); // Validasi response structure if (res.error) { throw new Error(res.error); } if (res && res.items && res.items.length > 0) { aiItems = res.items.map(i => { // Ensure proper structure return { item: String(i.name || i.item || 'Unknown'), name: String(i.name || i.item || 'Unknown'), quantity: Number(i.quantity || i.qty || 1) || 1, qty: Number(i.quantity || i.qty || 1) || 1, price: Number(i.price || 0) || 0, assignedTo: [] }; }); const sumItems = aiItems.reduce((s, i) => s + (i.price * i.qty), 0); aiDetails.subtotal = Number(res.subtotal) || sumItems; aiDetails.tax = Number(res.tax) || 0; aiDetails.service = Number(res.service) || 0; aiDetails.discount = Number(res.discount) || 0; aiGrandTotal = Number(res.grand_total) || sumItems; const calculatedTotal = aiDetails.subtotal + aiDetails.tax + aiDetails.service + aiDetails.discount; aiDetails.lainnya = aiGrandTotal - calculatedTotal; console.log('[OCR] Success! Items:', aiItems.length, 'Total:', aiGrandTotal); renderAiConfirm(); goToStep('stepAiConfirm'); } else { console.warn('[OCR] No items found in response'); showToast('Struk tidak terbaca. Coba foto yang lebih jelas.', 'error'); goToStep('stepMain'); } } catch (err) { console.error('[OCR] Critical error:', err.message); showToast('Gagal memproses struk: ' + err.message, 'error'); goToStep('stepMain'); } }
